## Feature 1 — Mermaid Diagram Rendering

### How the pipeline works today

```
MarkdownContent.svelte
  → remark/rehype pipeline (string → HTML)
  → renderedBlocks[]  (stable blocks cached by content hash)
  → mounted into containerRef  ←  DOM is here
  → rehypeEnhanceCodeBlocks adds copy/preview buttons to ALL <pre><code> blocks
```

Mermaid blocks arrive as `<pre><code class="language-mermaid">…</code></pre>` after the existing pipeline.

### Strategy

Do **not** add a new rehype plugin. Instead:

1. In `MarkdownContent.svelte`, add an `isStreaming` prop (default `false`).
2. Add a `$effect` that fires when `isStreaming` transitions from `true → false` (or when content settles on initial load) and scans `containerRef` for mermaid blocks.
3. For each found block:
   - Extract code text from `<code class="language-mermaid">`.
   - Call `mermaid.render(uniqueId, code)` (the async v11 API).
   - On success: replace the `<code>` inner content with the rendered SVG; add `data-mermaid-rendered="true"` to skip on subsequent effects.
   - On error: replace the `<code>` inner content with an error div; mark with `data-mermaid-error="true"`.
4. The copy button added by `rehypeEnhanceCodeBlocks` remains on the wrapper — it reads from the original `<code>` element's `data-code-id`, which still exists. Ensure the raw code text is preserved in `data-raw-code` so the copy handler can still access it.

Because `rehypeEnhanceCodeBlocks` stores code in the DOM via `data-code-id` referencing `window.__codeBlocks`, the raw text survives the SVG replacement as long as we don't remove `<code>`.  
→ Replace only the **inner content** of `<code>`, not the element itself.

### Mermaid initialization

```typescript
// src/lib/utils/mermaid.ts  (new file)
import mermaid from 'mermaid';

let initialized = false;

export function ensureMermaidInit(isDark: boolean) {
  mermaid.initialize({
    startOnLoad: false,
    theme: isDark ? 'dark' : 'default',
    securityLevel: 'loose',
  });
  initialized = true;
}

export async function renderMermaidInContainer(container: HTMLElement): Promise<void> {
  if (!initialized) return;

  const blocks = container.querySelectorAll<HTMLElement>(
    'pre code.language-mermaid:not([data-mermaid-rendered]):not([data-mermaid-error])'
  );

  for (const codeEl of blocks) {
    const code = codeEl.getAttribute('data-raw-code') ?? codeEl.textContent ?? '';
    const id = `mermaid-${Math.random().toString(36).slice(2)}`;

    try {
      const { svg } = await mermaid.render(id, code);
      // Preserve the text for copy-code, but hide it visually
      const rawSpan = document.createElement('span');
      rawSpan.style.display = 'none';
      rawSpan.textContent = code;

      const svgWrapper = document.createElement('div');
      svgWrapper.className = 'mermaid-diagram';
      svgWrapper.innerHTML = svg;

      codeEl.innerHTML = '';
      codeEl.appendChild(rawSpan);
      codeEl.appendChild(svgWrapper);
      codeEl.setAttribute('data-mermaid-rendered', 'true');
    } catch (_err) {
      const errorDiv = document.createElement('div');
      errorDiv.className = 'mermaid-error';
      errorDiv.textContent = 'Malformed mermaid syntax';
      codeEl.innerHTML = '';
      codeEl.appendChild(errorDiv);
      codeEl.setAttribute('data-mermaid-error', 'true');
    }
  }
}
```

## Feature 2 — "Mermaid diagram only" Filter

### Behaviour

Mirrors the existing `filterCodeblockOnly` pattern but targets mermaid blocks specifically.

- If one or more ` ```mermaid ` blocks are found → return only those blocks concatenated (stripped of surrounding text and other code blocks).
- If none found → return text unchanged.
- Applied at display time only (never persisted).

### `src/lib/utils/filters.ts` — add function

```typescript
export function applyMermaidOnly(text: string): string {
  const mermaidBlockRegex = /```mermaid\n[\s\S]*?```/g;
  const matches = text.match(mermaidBlockRegex);
  if (!matches || matches.length === 0) return text;
  return matches.join('\n\n');
}
```

## Feature 3 — Source/Citation Viewer (Rewrite)

### Current state

`ChatMessageLinks.svelte` renders an inline row of link chips. The `extractLinks()` utility parses both `[label](url)` markdown links and bare `https://` URLs. Both are already correct.

### New design

- Remove inline chip strip entirely.
- When `links.length > 0`: show a small **"N sources"** button (using `Link` icon) below the message content.
- Clicking opens a **shadcn-ui Dialog** listing each link with favicon, label, and URL.
- Each row has an external link icon to open in a new tab.

### Favicon fetching

```typescript
// src/lib/utils/favicon.ts  (new file)
const FAVICON_CACHE = new Map<string, string>();

export async function fetchFavicon(url: string): Promise<string> {
  const domain = new URL(url).hostname;
  if (FAVICON_CACHE.has(domain)) return FAVICON_CACHE.get(domain)!;

  // 1. Try Google favicon service
  const googleUrl = `https://www.google.com/s2/favicons?domain=${domain}&sz=16`;
  try {
    const res = await fetch(googleUrl, { signal: AbortSignal.timeout(3000) });
    if (res.ok) {
      const blob = await res.blob();
      const dataUrl = await blobToDataUrl(blob);
      FAVICON_CACHE.set(domain, dataUrl);
      return dataUrl;
    }
  } catch { /* fallthrough */ }

  // 2. Try root favicon.ico
  const rootFavicon = `${new URL(url).origin}/favicon.ico`;
  try {
    const res = await fetch(rootFavicon, { signal: AbortSignal.timeout(3000) });
    if (res.ok) {
      const blob = await res.blob();
      const dataUrl = await blobToDataUrl(blob);
      FAVICON_CACHE.set(domain, dataUrl);
      return dataUrl;
    }
  } catch { /* fallthrough */ }

  // 3. Fallback generic emoji
  FAVICON_CACHE.set(domain, '');
  return '';
}

async function blobToDataUrl(blob: Blob): Promise<string> {
  return new Promise((resolve) => {
    const reader = new FileReader();
    reader.onload = () => resolve(reader.result as string);
    reader.readAsDataURL(blob);
  });
}
```

## Feature 5 — Error Handling with Toast

### Current state

Network / timeout errors in `chat.service.ts` (line 327–358) build a `userFriendlyError` and call `onError(error)`. The `onError` callback chain ultimately triggers `DialogChatError` (a modal dialog). This is disruptive for transient errors.

The user wants a **toast** notification for:
- Network errors mid-stream (`TypeError` with `fetch` / `ECONNREFUSED`)
- Timeout errors (`ETIMEDOUT`)
- Additionally: errors that occur **mid-stream** (after generation has started)

`svelte-sonner`'s `toast` is already available in the project.

### Where to add toasts

**`src/lib/stores/chat.svelte.ts`** — find the `onError` handler that is called from the agentic flow and normal chat flow. Add toast calls for the two target error types.

The call chain is:
```
ChatScreen.svelte  →  chatStore.sendMessage(...)
  └→ chat.service.ts  (onError callback set here or passed)
       └→ if AbortError: silently ignored
          else: onError(userFriendlyError) called
               └→ chatStore.setError(...)  →  DialogChatError shown
```

Specifically in `src/lib/stores/chat.svelte.ts`, in the error handler block that currently calls `setError()` or equivalent, add before the existing dialog trigger:

```typescript
import { toast } from 'svelte-sonner';

// Inside the error handler:
if (error.name === 'NetworkError' || error.name === 'TimeoutError') {
  toast.error(error.message, {
    description: error.name === 'TimeoutError'
      ? 'The server took too long to respond. Check if it is still running.'
      : 'Could not reach the server. Check your connection or server status.',
    duration: 6000,
  });
  // Do NOT open DialogChatError for these transient errors — return early
  return;
}
// Existing error dialog path for other errors continues below
```

**Mid-stream errors** (the fetch stream breaks after some tokens arrive): these are currently caught inside the streaming response handler. Locate the stream reader `catch` block in `chat.service.ts` around `handleStreamResponse`, and apply the same toast pattern there.

### `todoStore` — `src/lib/stores/todo.svelte.ts` (NEW)

```typescript
import { DatabaseService } from '$lib/services/database.service';

export interface TodoItem {
  id: string;
  title: string;
  description?: string;
  completed: boolean;
  createdAt: number;
  updatedAt: number;
}

class TodoStore {
  items = $state<TodoItem[]>([]);
  conversationId = $state<string | null>(null);

  async loadForConversation(convId: string) {
    this.conversationId = convId;
    const conv = await DatabaseService.getConversation(convId);
    this.items = conv?.todos ?? [];
  }

  async execute(convId: string, action: string, args: Record<string, string>): Promise<object> {
    if (this.conversationId !== convId) await this.loadForConversation(convId);

    switch (action) {
      case 'create': {
        const item: TodoItem = {
          id: crypto.randomUUID(),
          title: args.title ?? 'Untitled',
          description: args.description,
          completed: false,
          createdAt: Date.now(),
          updatedAt: Date.now(),
        };
        this.items = [...this.items, item];
        await this.persist(convId);
        return { success: true, id: item.id };
      }
      case 'update': {
        this.items = this.items.map((i) =>
          i.id === args.id
            ? { ...i, title: args.title ?? i.title, description: args.description ?? i.description, updatedAt: Date.now() }
            : i
        );
        await this.persist(convId);
        return { success: true };
      }
      case 'complete': {
        this.items = this.items.map((i) =>
          i.id === args.id ? { ...i, completed: true, updatedAt: Date.now() } : i
        );
        await this.persist(convId);
        return { success: true };
      }
      case 'delete': {
        this.items = this.items.filter((i) => i.id !== args.id);
        await this.persist(convId);
        return { success: true };
      }
      case 'list': {
        return { success: true, items: this.items };
      }
      default:
        return { success: false, error: `Unknown action: ${action}` };
    }
  }

  private async persist(convId: string) {
    await DatabaseService.updateConversation(convId, { todos: this.items });
  }
}

export const todoStore = new TodoStore();
```
